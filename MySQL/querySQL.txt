    public function createSQL($array)
    {
        $sql = "";
        foreach( $array as $index => $arr){
            if(!empty($array[$index]) && empty($sql)){
                $sql .= $index.' like \'%'.$array[$index].'%\'';
            }else if(!empty($array[$index]) && !empty($sql)){
                $sql .= ' and '.$index.' like \'%'.$array[$index].'%\'';
            }
        }
        return $sql;
    }
